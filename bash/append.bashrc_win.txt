# the 2 branches susceptible to conflict

}
gg(){ 
  cnt=3
  branch=$1
  if [[ "$1" =~ ^[0-9]+$ ]]; then # note no quote around regex
    cnt=$1 #by default shows that many commits
    branch=$2
  fi
  git log -$cnt $branch
}
git-showtool(){
  ref=${1:-HEAD}
  git show --name-status $ref
  set -x
  git difftool $ref~  $ref
  { set +x; } 2>/dev/null
}
